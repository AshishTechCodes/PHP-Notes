Complete SQL Reference Guide

All SQL concepts with practical examples

SQL Language Types

Data Query Language (DQL)

Used to query and retrieve data from the database.

Main Command: SELECT

-- Basic SELECT
SELECT * FROM employees;

-- SELECT specific columns
SELECT first_name, last_name, salary FROM employees;

-- SELECT with WHERE condition
SELECT * FROM products WHERE price > 100;

-- SELECT with ORDER BY
SELECT * FROM customers ORDER BY last_name ASC;

-- SELECT with LIMIT
SELECT * FROM orders LIMIT 10;

Data Definition Language (DDL)

Used to define and manage database structures.

  • CREATE: Creates new database objects
  • ALTER: Modifies existing objects
  • DROP: Deletes objects
  • TRUNCATE/DELETE: Removes records but retains structure
-- CREATE TABLE
CREATE TABLE employees (
    id INT PRIMARY KEY,
    name VARCHAR(100),
    salary DECIMAL(10,2),
    hire_date DATE
);

-- ALTER TABLE (add column)
ALTER TABLE employees ADD COLUMN department VARCHAR(50);

-- ALTER TABLE (modify column)
ALTER TABLE employees MODIFY COLUMN name VARCHAR(150);

-- DROP TABLE
DROP TABLE temp_employees;

-- TRUNCATE TABLE
TRUNCATE TABLE audit_log;

-- DELETE specific rows
DELETE FROM employees WHERE id = 101;

Data Control Language (DCL)

Used to control access and permissions.

  • GRANT: Assigns privileges
  • REVOKE: Removes privileges
-- GRANT privileges
GRANT SELECT, INSERT ON employees TO user1;

-- GRANT all privileges
GRANT ALL PRIVILEGES ON database.* TO 'admin'@'localhost';

-- REVOKE privileges
REVOKE DELETE ON employees FROM user2;

-- REVOKE all privileges
REVOKE ALL PRIVILEGES ON database.* FROM 'user'@'localhost';

-- Create a new user
CREATE USER 'newuser'@'localhost' IDENTIFIED BY 'password';

-- Grant specific database access
GRANT SELECT ON mydb.* TO 'newuser'@'localhost';

Transaction Control Language (TCL)

Manages changes and ensures data integrity.

  • COMMIT: Saves changes
  • ROLLBACK: Undoes changes
  • SAVEPOINT: Sets rollback points
  • SET TRANSACTION: Configures transactions
-- Simple transaction
BEGIN TRANSACTION;
UPDATE accounts SET balance = balance - 100 WHERE id = 1;
UPDATE accounts SET balance = balance + 100 WHERE id = 2;
COMMIT;

-- Transaction with ROLLBACK
BEGIN TRANSACTION;
INSERT INTO orders (customer_id, amount) VALUES (5, 99.99);
-- If something goes wrong:
ROLLBACK;

-- Transaction with SAVEPOINT
BEGIN TRANSACTION;
UPDATE products SET stock = stock - 1 WHERE id = 10;
SAVEPOINT after_product_update;
INSERT INTO order_details (order_id, product_id) VALUES (1001, 10);
-- If order details fails but product update is OK:
ROLLBACK TO after_product_update;
COMMIT;

-- SET TRANSACTION
SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
BEGIN TRANSACTION;
-- Your SQL statements here
COMMIT;

Arithmetic Operations in SQL

SQL supports standard arithmetic operations in queries:

Addition (+)

-- Add a fixed amount
SELECT salary, salary + 5000 AS increased_salary FROM employees;

-- Add two columns
SELECT quantity, unit_price, quantity + unit_price AS total FROM order_items;

-- Add with NULL handling (COALESCE)
SELECT price, discount, price + COALESCE(discount, 0) AS final_price FROM products;

Subtraction (-)

-- Subtract a fixed amount
SELECT salary, salary - tax AS net_salary FROM employees;

-- Subtract two columns
SELECT revenue, expenses, revenue - expenses AS profit FROM financials;

-- Date subtraction (days between dates)
SELECT order_date, ship_date, ship_date - order_date AS days_to_ship FROM orders;

Multiplication (*)

-- Multiply two columns
SELECT quantity, price, quantity * price AS total FROM order_items;

-- Multiply with percentage
SELECT salary, salary * 1.1 AS salary_with_10pct_raise FROM employees;

-- Multiply with NULL handling
SELECT hours_worked, rate, hours_worked * COALESCE(rate, 0) AS earnings FROM timesheets;

Division (/)

-- Divide two columns
SELECT revenue, employees, revenue/employees AS rev_per_emp FROM departments;

-- Division with type casting
SELECT quantity, total, CAST(total AS FLOAT)/quantity AS unit_price FROM sales;

-- Safe division (avoid divide by zero)
SELECT a, b, CASE WHEN b = 0 THEN NULL ELSE a/b END AS ratio FROM calculations;

Modulus (%)

-- Find even IDs
SELECT id FROM orders WHERE id % 2 = 0;

-- Group data into buckets
SELECT value, value % 10 AS bucket FROM measurements;

-- Check divisibility
SELECT number, CASE WHEN number % 5 = 0 THEN 'Divisible by 5' ELSE 'Not divisible' END FROM numbers;

Combined Operations

-- Complex calculation
SELECT quantity, price, discount, 
       (quantity * price) - (quantity * price * discount/100) AS net_total 
FROM order_items;

-- Using parentheses for order of operations
SELECT (revenue - expenses) / employees AS profit_per_employee FROM departments;

-- Mathematical expressions in WHERE
SELECT * FROM products WHERE (price * 0.9) < 50;

Set Operators in SQL

UNION

Combines result sets of two or more SELECT queries and removes duplicate rows.

Note: All SELECT statements must have the same number of columns with compatible data types.

-- Combine current and discontinued products
SELECT product_id, product_name FROM current_products
UNION
SELECT product_id, product_name FROM discontinued_products;

-- UNION with WHERE clauses
SELECT customer_id FROM online_orders WHERE order_date > '2023-01-01'
UNION
SELECT customer_id FROM in_store_orders WHERE order_date > '2023-01-01';

-- UNION with ORDER BY (applies to entire result)
SELECT name, 'Employee' AS type FROM employees
UNION
SELECT name, 'Customer' AS type FROM customers
ORDER BY name;

UNION ALL

Combines result sets but keeps all rows including duplicates.

-- Combine all orders including duplicates
SELECT customer_id FROM online_orders
UNION ALL
SELECT customer_id FROM in_store_orders;

-- UNION ALL is faster than UNION when duplicates don't matter
SELECT product_id FROM warehouse1_inventory
UNION ALL
SELECT product_id FROM warehouse2_inventory;

-- UNION ALL with different numbers of rows
SELECT 'A' AS letter, 1 AS num UNION ALL SELECT 'B', 2 UNION ALL SELECT 'B', 2;

INTERSECT

Returns only rows that exist in both queries.

Note: Not supported in all database systems (MySQL doesn't support INTERSECT directly).

-- Employees who are also managers
SELECT employee_id FROM full_time_employees
INTERSECT
SELECT employee_id FROM managers;

-- Products ordered in both January and February
SELECT product_id FROM january_orders
INTERSECT
SELECT product_id FROM february_orders;

-- MySQL alternative using INNER JOIN
SELECT DISTINCT e.employee_id 
FROM full_time_employees e
INNER JOIN managers m ON e.employee_id = m.employee_id;

EXCEPT (or MINUS)

Returns rows from the first query not present in the second.

Note: Called MINUS in Oracle, EXCEPT in SQL Server and PostgreSQL.

-- Customers who haven't placed orders
SELECT customer_id FROM all_customers
EXCEPT
SELECT customer_id FROM premium_customers;

-- Products never ordered
SELECT product_id FROM products
EXCEPT
SELECT product_id FROM order_details;

-- MySQL alternative using LEFT JOIN
SELECT c.customer_id 
FROM all_customers c
LEFT JOIN premium_customers p ON c.customer_id = p.customer_id
WHERE p.customer_id IS NULL;

Logical Operators in SQL

Used to combine multiple conditions in WHERE or HAVING clauses.

AND Operator

Returns true if all conditions are true.

-- Multiple conditions with AND
SELECT * FROM employees
WHERE department = 'Sales' AND salary > 50000;

-- AND with different column conditions
SELECT * FROM products
WHERE category = 'Electronics' AND price < 1000 AND stock_quantity > 0;

-- AND with date range
SELECT * FROM orders
WHERE order_date >= '2023-01-01' AND order_date <= '2023-01-31';

-- Combining AND with other operators
SELECT * FROM customers
WHERE status = 'Active' AND (last_purchase_date > '2023-01-01' OR total_purchases > 1000);

OR Operator

Returns true if at least one condition is true.

-- Multiple conditions with OR
SELECT * FROM products
WHERE category = 'Electronics' OR price > 1000;

-- OR with different column conditions
SELECT * FROM employees
WHERE department = 'HR' OR department = 'Finance' OR salary > 80000;

-- OR with IN (alternative)
SELECT * FROM customers
WHERE state = 'CA' OR state = 'NY' OR state = 'TX';
-- Equivalent to:
SELECT * FROM customers WHERE state IN ('CA', 'NY', 'TX');

-- Combining OR with AND (use parentheses)
SELECT * FROM orders
WHERE (status = 'Shipped' OR status = 'Delivered') AND total_amount > 100;

NOT Operator

Reverses the logical state of a condition.

-- Simple NOT condition
SELECT * FROM customers
WHERE NOT country = 'USA';

-- NOT with IN
SELECT * FROM products
WHERE NOT category IN ('Electronics', 'Appliances');

-- NOT with LIKE
SELECT * FROM employees
WHERE NOT first_name LIKE 'A%';

-- NOT with IS NULL
SELECT * FROM orders
WHERE NOT shipping_date IS NULL;

-- NOT with EXISTS
SELECT * FROM departments d
WHERE NOT EXISTS (SELECT 1 FROM employees e WHERE e.department_id = d.id);

SQL WHERE Clause

Filters records based on specified conditions.

Simple Comparison

-- Greater than
SELECT first_name, last_name, salary  
FROM employees  
WHERE salary > 50000;

-- Less than or equal to
SELECT product_name, price
FROM products
WHERE price <= 100;

-- Not equal (both != and <> work)
SELECT order_id, status
FROM orders
WHERE status != 'Cancelled';

SELECT order_id, status
FROM orders
WHERE status <> 'Cancelled';

BETWEEN for a Range

-- Numeric range
SELECT first_name, last_name, salary  
FROM employees  
WHERE salary BETWEEN 40000 AND 60000;

-- Date range
SELECT order_id, order_date, total_amount
FROM orders
WHERE order_date BETWEEN '2023-01-01' AND '2023-01-31';

-- NOT BETWEEN
SELECT product_name, price
FROM products
WHERE price NOT BETWEEN 10 AND 100;

-- BETWEEN with strings
SELECT product_name
FROM products
WHERE product_name BETWEEN 'A' AND 'D';

IN for Multiple Values

-- Simple IN
SELECT first_name, last_name  
FROM employees  
WHERE department_id IN (1, 3, 5);

-- IN with subquery
SELECT customer_name
FROM customers
WHERE customer_id IN (SELECT customer_id FROM orders WHERE total_amount > 1000);

-- NOT IN
SELECT product_name
FROM products
WHERE category_id NOT IN (5, 7, 9);

-- IN with NULL values (careful with this)
SELECT employee_name
FROM employees
WHERE department_id IN (1, 3, NULL); -- NULL values won't match

LIKE for Pattern Matching

-- Starts with 'A'
SELECT first_name, last_name  
FROM employees  
WHERE first_name LIKE 'A%';

-- Ends with 'son'
SELECT last_name
FROM employees
WHERE last_name LIKE '%son';

-- Contains 'tech'
SELECT product_name
FROM products
WHERE product_name LIKE '%tech%';

-- Exactly 5 characters starting with 'S'
SELECT first_name
FROM employees
WHERE first_name LIKE 'S____';

-- NOT LIKE
SELECT email
FROM customers
WHERE email NOT LIKE '%@gmail.com';

-- LIKE with escape character
SELECT description
FROM products
WHERE description LIKE '%50%%' ESCAPE ''; -- Finds entries with '50%'

IS NULL for NULL Values

-- Simple NULL check
SELECT first_name, last_name  
FROM employees  
WHERE department_id IS NULL;

-- IS NOT NULL
SELECT order_id, customer_id
FROM orders
WHERE shipping_date IS NOT NULL;

-- NULL in WHERE with other conditions
SELECT product_name, price
FROM products
WHERE price IS NULL OR price > 100;

-- NULL with NOT IN (careful with this)
SELECT employee_name
FROM employees
WHERE department_id NOT IN (SELECT dept_id FROM departments WHERE active = 1);
-- This won't return rows where department_id IS NULL

Complex Conditions

-- Combining multiple conditions
SELECT product_name, price, stock_quantity
FROM products
WHERE (category = 'Electronics' OR category = 'Computers')
  AND price BETWEEN 100 AND 1000
  AND stock_quantity > 0;

-- Using parentheses to group conditions
SELECT order_id, customer_id, total_amount
FROM orders
WHERE (status = 'Shipped' AND shipping_date IS NOT NULL)
   OR (status = 'Processing' AND order_date > '2023-01-01');

-- Nested conditions
SELECT employee_name, salary, department
FROM employees
WHERE department = 'Sales' 
  AND (salary > 50000 OR 
      (hire_date < '2020-01-01' AND salary > 40000));

Aggregate Functions in SQL

Functions that operate on collections of values and return a single result.

COUNT()

Returns the number of rows that match a condition.

-- Count all rows
SELECT COUNT(*) FROM employees;

-- Count non-NULL values in a column
SELECT COUNT(department_id) FROM employees WHERE salary > 50000;

-- Count distinct values
SELECT COUNT(DISTINCT department) FROM employees;

-- Count with GROUP BY
SELECT department, COUNT(*) AS employee_count 
FROM employees 
GROUP BY department;

-- Count with HAVING
SELECT department, COUNT(*) 
FROM employees 
GROUP BY department 
HAVING COUNT(*) > 5;

SUM()

Returns the total sum of a numeric column.

-- Simple SUM
SELECT SUM(salary) FROM employees;

-- SUM with WHERE
SELECT SUM(sales) FROM sales_data WHERE region = 'West';

-- SUM with GROUP BY
SELECT department, SUM(sales) 
FROM sales_data 
GROUP BY department;

-- SUM of expression
SELECT SUM(quantity * unit_price) AS total_sales 
FROM order_items;

-- SUM with DISTINCT
SELECT SUM(DISTINCT salary) FROM employees;

AVG()

Returns the average value of a numeric column.

-- Simple average
SELECT AVG(salary) FROM employees;

-- AVG with WHERE
SELECT AVG(price) FROM products WHERE category = 'Electronics';

-- AVG with GROUP BY
SELECT department, AVG(salary) 
FROM employees 
GROUP BY department;

-- AVG with ROUND
SELECT ROUND(AVG(price), 2) AS avg_price FROM products;

-- AVG with NULL handling
SELECT AVG(COALESCE(discount, 0)) AS avg_discount FROM orders;

MIN()

Returns the minimum value of a column.

-- Simple MIN
SELECT MIN(salary) FROM employees;

-- MIN with WHERE
SELECT MIN(price) FROM products WHERE stock_quantity > 0;

-- MIN with GROUP BY
SELECT product_category, MIN(price) 
FROM products 
GROUP BY product_category;

-- MIN with dates
SELECT MIN(hire_date) FROM employees;

-- MIN with strings
SELECT MIN(product_name) FROM products; -- Alphabetical first

MAX()

Returns the maximum value of a column.

-- Simple MAX
SELECT MAX(salary) FROM employees;

-- MAX with WHERE
SELECT MAX(price) FROM products WHERE category = 'Electronics';

-- MAX with GROUP BY
SELECT department, MAX(sales) 
FROM sales_data 
GROUP BY department;

-- MAX with dates
SELECT MAX(order_date) FROM orders;

-- MAX with strings
SELECT MAX(product_name) FROM products; -- Alphabetical last

GROUP_CONCAT() (MySQL)

Concatenates values from multiple rows into a string.

-- Simple concatenation
SELECT department, 
       GROUP_CONCAT(employee_name SEPARATOR ', ') 
FROM employees 
GROUP BY department;

-- With ORDER BY
SELECT department,
       GROUP_CONCAT(employee_name ORDER BY hire_date SEPARATOR ' | ')
FROM employees
GROUP BY department;

-- With DISTINCT
SELECT project_id,
       GROUP_CONCAT(DISTINCT employee_id ORDER BY employee_id)
FROM project_assignments
GROUP BY project_id;

-- Limited number of items
SELECT category,
       GROUP_CONCAT(product_name LIMIT 5)
FROM products
GROUP BY category;

-- Alternative in SQL Server: STRING_AGG
-- SELECT department, STRING_AGG(employee_name, ', ') FROM employees GROUP BY department;

GROUP BY with HAVING

GROUP BY groups rows with the same values, and HAVING filters groups after aggregation.

-- Basic GROUP BY
SELECT department, AVG(salary) as avg_salary
FROM employees
GROUP BY department;

-- Multiple GROUP BY columns
SELECT department, job_title, COUNT(*) as employee_count
FROM employees
GROUP BY department, job_title;

-- HAVING to filter groups
SELECT department, AVG(salary) as avg_salary
FROM employees
GROUP BY department
HAVING AVG(salary) > 50000;

-- HAVING with multiple conditions
SELECT customer_id, COUNT(*) as order_count, SUM(total_amount) as total_spent
FROM orders
GROUP BY customer_id
HAVING COUNT(*) > 5 AND SUM(total_amount) > 1000;

-- WHERE vs HAVING
SELECT department, AVG(salary) as avg_salary
FROM employees
WHERE hire_date > '2020-01-01'  -- Filters rows before grouping
GROUP BY department
HAVING AVG(salary) > 60000;     -- Filters groups after aggregation

SQL Key Constraints

Rules that ensure data integrity and consistency.

PRIMARY KEY

Uniquely identifies each record in a table. No duplicates or NULLs allowed.

-- Single column primary key
CREATE TABLE employees (
    emp_id INT PRIMARY KEY,
    name VARCHAR(100),
    department VARCHAR(50)
);

-- Primary key with auto-increment
CREATE TABLE products (
    product_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    price DECIMAL(10,2)
);

-- Composite primary key (multiple columns)
CREATE TABLE order_items (
    order_id INT,
    product_id INT,
    quantity INT,
    PRIMARY KEY (order_id, product_id)
);

-- Add primary key to existing table
ALTER TABLE customers ADD PRIMARY KEY (customer_id);

-- Drop primary key
ALTER TABLE employees DROP PRIMARY KEY;

FOREIGN KEY

Establishes relationships between tables.

-- Basic foreign key
CREATE TABLE orders (
    order_id INT PRIMARY KEY,
    customer_id INT,
    order_date DATE,
    FOREIGN KEY (customer_id) REFERENCES customers(customer_id)
);

-- Foreign key with ON DELETE CASCADE
CREATE TABLE order_items (
    item_id INT PRIMARY KEY,
    order_id INT,
    product_id INT,
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE
);

-- Multiple foreign keys
CREATE TABLE employee_projects (
    emp_id INT,
    project_id INT,
    hours_worked INT,
    FOREIGN KEY (emp_id) REFERENCES employees(emp_id),
    FOREIGN KEY (project_id) REFERENCES projects(project_id)
);

-- Add foreign key to existing table
ALTER TABLE orders
ADD CONSTRAINT fk_customer
FOREIGN KEY (customer_id) REFERENCES customers(customer_id);

-- Disable foreign key checks temporarily
SET FOREIGN_KEY_CHECKS = 0;
-- Perform operations that might violate constraints
SET FOREIGN_KEY_CHECKS = 1;

UNIQUE

Ensures all values in a column are unique.

-- Single column unique constraint
CREATE TABLE users (
    user_id INT PRIMARY KEY,
    username VARCHAR(50) UNIQUE,
    email VARCHAR(100) UNIQUE
);

-- Composite unique constraint (combination must be unique)
CREATE TABLE employee_departments (
    emp_id INT,
    dept_id INT,
    start_date DATE,
    UNIQUE (emp_id, dept_id)
);

-- Add unique constraint to existing table
ALTER TABLE products ADD CONSTRAINT uq_product_code UNIQUE (product_code);

-- Unique constraint with NULL values
CREATE TABLE contacts (
    contact_id INT PRIMARY KEY,
    email VARCHAR(100) UNIQUE, -- Only one NULL allowed in some databases
    phone VARCHAR(20) UNIQUE
);

-- Drop unique constraint
ALTER TABLE users DROP CONSTRAINT username;

DEFAULT

Provides a default value for a column.

-- Basic default values
CREATE TABLE products (
    product_id INT PRIMARY KEY,
    name VARCHAR(100),
    price DECIMAL(10,2) DEFAULT 0.00,
    in_stock BOOLEAN DEFAULT TRUE
);

-- Default with CURRENT_TIMESTAMP
CREATE TABLE orders (
    order_id INT PRIMARY KEY,
    order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(20) DEFAULT 'Pending'
);

-- Default with functions
CREATE TABLE users (
    user_id INT PRIMARY KEY,
    username VARCHAR(50),
    created_at DATETIME DEFAULT NOW(),
    updated_at DATETIME DEFAULT NOW() ON UPDATE NOW()
);

-- Add default to existing column
ALTER TABLE employees ALTER COLUMN vacation_days SET DEFAULT 10;

-- Remove default
ALTER TABLE products ALTER COLUMN price DROP DEFAULT;

CHECK

Ensures column values satisfy a condition.

-- Simple check constraint
CREATE TABLE employees (
    emp_id INT PRIMARY KEY,
    name VARCHAR(100),
    salary DECIMAL(10,2) CHECK (salary > 0),
    age INT CHECK (age >= 18)
);

-- Named check constraint
CREATE TABLE products (
    product_id INT PRIMARY KEY,
    name VARCHAR(100),
    price DECIMAL(10,2),
    discount DECIMAL(10,2),
    CONSTRAINT chk_price CHECK (price > 0),
    CONSTRAINT chk_discount CHECK (discount >= 0 AND discount <= price)
);

-- Check with function
CREATE TABLE reservations (
    res_id INT PRIMARY KEY,
    start_date DATE,
    end_date DATE,
    CONSTRAINT chk_dates CHECK (end_date >= start_date)
);

-- Add check to existing table
ALTER TABLE employees ADD CONSTRAINT chk_salary CHECK (salary > 0);

-- Disable check constraints temporarily (SQL Server)
ALTER TABLE products NOCHECK CONSTRAINT chk_price;
-- Re-enable
ALTER TABLE products CHECK CONSTRAINT chk_price;

NOT NULL

Ensures a column cannot have NULL values.

-- Basic NOT NULL
CREATE TABLE customers (
    customer_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL
);

-- NOT NULL with DEFAULT
CREATE TABLE orders (
    order_id INT PRIMARY KEY,
    order_date DATE NOT NULL DEFAULT CURRENT_DATE,
    status VARCHAR(20) NOT NULL DEFAULT 'Pending'
);

-- Add NOT NULL to existing column
ALTER TABLE employees MODIFY COLUMN hire_date DATE NOT NULL;

-- Remove NOT NULL constraint
ALTER TABLE products MODIFY COLUMN description VARCHAR(500) NULL;

-- NOT NULL with CHECK
CREATE TABLE accounts (
    account_id INT PRIMARY KEY,
    balance DECIMAL(15,2) NOT NULL CHECK (balance >= 0)
);

-- Temporarily disable NOT NULL (PostgreSQL)
SET CONSTRAINTS ALL DEFERRED;
-- Perform operations
SET CONSTRAINTS ALL IMMEDIATE;

SQL JOINS

Combine rows from two or more tables based on related columns.

INNER JOIN

Returns only rows with matching values in both tables.

-- Basic INNER JOIN
SELECT employees.name, departments.department_name
FROM employees
INNER JOIN departments ON employees.dept_id = departments.dept_id;

-- INNER JOIN with WHERE
SELECT o.order_id, c.customer_name, o.order_date
FROM orders o
INNER JOIN customers c ON o.customer_id = c.customer_id
WHERE o.order_date > '2023-01-01';

-- INNER JOIN with multiple tables
SELECT p.product_name, c.category_name, s.supplier_name
FROM products p
INNER JOIN categories c ON p.category_id = c.category_id
INNER JOIN suppliers s ON p.supplier_id = s.supplier_id;

-- INNER JOIN with GROUP BY
SELECT d.department_name, COUNT(e.emp_id) AS employee_count
FROM departments d
INNER JOIN employees e ON d.dept_id = e.dept_id
GROUP BY d.department_name;

-- INNER JOIN with aliases
SELECT e.name, d.department_name, j.title
FROM employees e
INNER JOIN departments d ON e.dept_id = d.dept_id
INNER JOIN jobs j ON e.job_id = j.job_id;

LEFT JOIN (LEFT OUTER JOIN)

Returns all rows from the left table and matched rows from the right table.

-- Basic LEFT JOIN
SELECT customers.customer_name, orders.order_date
FROM customers
LEFT JOIN orders ON customers.customer_id = orders.customer_id;

-- LEFT JOIN with WHERE to find non-matches
SELECT c.customer_name
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id
WHERE o.order_id IS NULL;

-- LEFT JOIN with multiple conditions
SELECT e.employee_name, p.project_name
FROM employees e
LEFT JOIN employee_projects ep ON e.emp_id = ep.emp_id AND ep.status = 'Active'
LEFT JOIN projects p ON ep.project_id = p.project_id;

-- LEFT JOIN with aggregate functions
SELECT d.department_name, COUNT(e.emp_id) AS employee_count
FROM departments d
LEFT JOIN employees e ON d.dept_id = e.dept_id
GROUP BY d.department_name;

-- LEFT JOIN with COALESCE to handle NULLs
SELECT p.product_name, COALESCE(SUM(oi.quantity), 0) AS total_ordered
FROM products p
LEFT JOIN order_items oi ON p.product_id = oi.product_id
GROUP BY p.product_name;

RIGHT JOIN (RIGHT OUTER JOIN)

Returns all rows from the right table and matched rows from the left table.

-- Basic RIGHT JOIN
SELECT e.employee_name, d.department_name
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.dept_id;

-- RIGHT JOIN to find departments without employees
SELECT d.department_name
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.dept_id
WHERE e.emp_id IS NULL;

-- RIGHT JOIN with multiple tables
SELECT c.category_name, p.product_name
FROM products p
RIGHT JOIN categories c ON p.category_id = c.category_id
RIGHT JOIN suppliers s ON p.supplier_id = s.supplier_id;

-- RIGHT JOIN with aggregate functions
SELECT s.supplier_name, COUNT(p.product_id) AS product_count
FROM products p
RIGHT JOIN suppliers s ON p.supplier_id = s.supplier_id
GROUP BY s.supplier_name;

-- RIGHT JOIN with COALESCE
SELECT c.category_name, COALESCE(COUNT(p.product_id), 0) AS product_count
FROM products p
RIGHT JOIN categories c ON p.category_id = c.category_id
GROUP BY c.category_name;

FULL OUTER JOIN

Returns all rows when there is a match in either left or right table.

Note: MySQL doesn't support FULL OUTER JOIN directly. Use UNION of LEFT and RIGHT JOINs.

-- Basic FULL OUTER JOIN (SQL Server, PostgreSQL)
SELECT e.employee_name, d.department_name
FROM employees e
FULL OUTER JOIN departments d ON e.dept_id = d.dept_id;

-- FULL OUTER JOIN to see all relationships
SELECT c.customer_name, o.order_id, o.order_date
FROM customers c
FULL OUTER JOIN orders o ON c.customer_id = o.customer_id;

-- MySQL alternative using UNION
SELECT e.employee_name, d.department_name
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.dept_id
UNION
SELECT e.employee_name, d.department_name
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.dept_id
WHERE e.emp_id IS NULL;

-- FULL OUTER JOIN with WHERE
SELECT p.product_name, s.supplier_name
FROM products p
FULL OUTER JOIN suppliers s ON p.supplier_id = s.supplier_id
WHERE p.discontinued = 0 OR p.discontinued IS NULL;

CROSS JOIN

Returns Cartesian product of two tables.

-- Basic CROSS JOIN
SELECT p.product_name, s.size
FROM products p
CROSS JOIN sizes s;

-- CROSS JOIN for generating test data
SELECT u.username, p.permission
FROM users u
CROSS JOIN permissions p
WHERE u.role = 'Admin';

-- CROSS JOIN with WHERE (similar to INNER JOIN)
SELECT e.employee_name, d.department_name
FROM employees e
CROSS JOIN departments d
WHERE e.dept_id = d.dept_id;

-- CROSS JOIN for date ranges
SELECT d.date, r.room_number
FROM dates d
CROSS JOIN rooms r
WHERE d.date BETWEEN '2023-01-01' AND '2023-01-07';

-- CROSS JOIN with LIMIT
SELECT c.color, p.product_name
FROM colors c
CROSS JOIN products p
LIMIT 100;

SELF JOIN

Joins a table with itself.

-- Basic SELF JOIN (employee-manager relationship)
SELECT e.employee_name, m.employee_name AS manager_name
FROM employees e
LEFT JOIN employees m ON e.manager_id = m.emp_id;

-- SELF JOIN for finding pairs
SELECT a.product_name, b.product_name
FROM products a
JOIN products b ON a.category_id = b.category_id AND a.product_id < b.product_id;

-- SELF JOIN for hierarchical data
SELECT c1.category_name, c2.category_name AS parent_category
FROM categories c1
LEFT JOIN categories c2 ON c1.parent_id = c2.category_id;

-- SELF JOIN for comparing rows
SELECT a.order_id, a.order_date, b.order_date AS previous_order
FROM orders a
JOIN orders b ON a.customer_id = b.customer_id 
  AND b.order_date < a.order_date
  AND DATEDIFF(a.order_date, b.order_date) <= 30;

-- SELF JOIN with aggregate
SELECT d1.department_name, COUNT(d2.employee_id) AS employee_count
FROM departments d1
LEFT JOIN departments d2 ON d1.dept_id = d2.dept_id
GROUP BY d1.department_name;

SQL Aliases

Temporary names for tables or columns in queries.

Column Aliases

Temporary names for columns in result sets.

-- Basic column aliases
SELECT 
    first_name AS "First Name", 
    last_name AS "Last Name",
    salary * 12 AS "Annual Salary"
FROM employees;

-- Aliases with aggregate functions
SELECT 
    department,
    COUNT(*) AS employee_count,
    AVG(salary) AS avg_salary,
    MAX(salary) AS max_salary
FROM employees
GROUP BY department;

-- Aliases with calculations
SELECT 
    product_name,
    unit_price,
    quantity,
    unit_price * quantity AS total_price,
    (unit_price * quantity) * 0.1 AS discount
FROM order_items;

-- Aliases with CASE expressions
SELECT 
    product_name,
    CASE 
        WHEN price > 1000 THEN 'Premium'
        WHEN price > 100 THEN 'Standard'
        ELSE 'Budget'
    END AS price_category
FROM products;

-- Aliases with string functions
SELECT 
    CONCAT(first_name, ' ', last_name) AS full_name,
    LENGTH(email) AS email_length,
    SUBSTRING(phone, 1, 3) AS area_code
FROM customers;

Table Aliases

Temporary names for tables, especially in joins.

-- Basic table aliases
SELECT e.name, d.department_name
FROM employees AS e
INNER JOIN departments AS d ON e.dept_id = d.dept_id;

-- Shorter aliases (AS keyword is optional)
SELECT o.order_id, c.customer_name, o.order_date
FROM orders o
JOIN customers c ON o.customer_id = c.customer_id;

-- Aliases with self joins
SELECT 
    emp.name AS employee_name,
    mgr.name AS manager_name
FROM employees emp
LEFT JOIN employees mgr ON emp.manager_id = mgr.emp_id;

-- Aliases with subqueries
SELECT 
    d.department_name,
    emp_count.employee_count
FROM departments d
JOIN (
    SELECT dept_id, COUNT(*) AS employee_count
    FROM employees
    GROUP BY dept_id
) emp_count ON d.dept_id = emp_count.dept_id;

-- Aliases with multiple joins
SELECT 
    o.order_id,
    c.customer_name,
    p.product_name,
    oi.quantity,
    oi.unit_price
FROM orders o
JOIN customers c ON o.customer_id = c.customer_id
JOIN order_items oi ON o.order_id = oi.order_id
JOIN products p ON oi.product_id = p.product_id;
